### 多级指针
```c
#include <stdio.h>

int main() {

    int num = 999;

        int * num_p = &num; // 取出num的内存地址（假设1000H）赋给num_p，一级指针

    int ** num_p_p = &num_p; // 取出num_p的内存地址（假设2000H）赋给num_p_p，二级指针

    int *** num_ppp = &num_p_p; // 取出num_p_p的内存地址（假设3000H）赋给num_ppp，三级指针

	// 取出num_p指向的地址（即1000H）所存的值，得到999
    printf("获取最终的值：%d\n", *num_p); 

	// 右起第一个*，取出num_p_p指向的地址（即2000H）所存的值，得到1000H；右起第二个*，取出地址1000H所存的值，得到999
    printf("获取最终的值：%d\n", **num_p_p); 

	// 右起第一个*，取出num_ppp指向的地址（即3000H）所存的值，得到1000H；右起第二个*，取出num_p_p指向的地址（即2000H）所存的值，得到1000H；右起第三个*，取出地址1000H所存的值，得到999
    printf("获取最终的值：%d\n", *** num_ppp);

    // 多级指针的意义是什么啊
    // test1(num_p_p)

    // printf("num_p的自己的内存地址是：%p， num_p_p自己的额内存地址是：%p\n", &num_p, &num_p_p);

    // *num_ppp  取的是什么？

    return 0;
}
```
